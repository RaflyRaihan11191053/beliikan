# Create Product
Request :
  - Method : POST
  - Endpoint : /api/product
  - Header : 
      - Content-Type: application/json
      - Accept: application/json
      
  - Body :
 
  ```json 
{
    "name" : "string",
    "price" : "long",
    "description" : "text"
    "stock" : "integer"
    "img_url" : "string"
}
```
Respons :

```json 
{
    "code" : "number",
    "status" : "string",
    "data" : {
         "name" : "string",
         "price" : "long",
         "description" : "text"
         "stock" : "integer"
         "img_url" : "string",
         "createdAt" : "date",
         "updatedAt" : "date"
     }
}
```
